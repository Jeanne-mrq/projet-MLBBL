plot_quantitative <- function(df) {
  vars_num <- names(df)[sapply(df, is.numeric)]
  plots <- lapply(vars_num, function(v) {
    ggplot(df, aes(x = .data[[v]])) +
      geom_histogram(aes(y = after_stat(density)), bins = 30,
                     fill = "grey80", color = "black") +
      geom_density(alpha = 0.2, color = "steelblue") +
      labs(title = v, x = v, y = "Densité") +
      theme_minimal()
  })
  do.call(grid.arrange, c(plots, ncol = 3))
}

metriques_reg <- function(obs, pred, modele = "", lm_model = NULL) {
  rmse <- sqrt(mean((obs - pred)^2))
  mae  <- mean(abs(obs - pred))
  if (!is.null(lm_model)) {
    s      <- summary(lm_model)
    r2     <- s$r.squared
    r2_adj <- s$adj.r.squared
    mse    <- s$sigma^2
  } else {
    n      <- length(obs)
    r2     <- 1 - sum((obs - pred)^2) / sum((obs - mean(obs))^2)
    r2_adj <- 1 - ((1 - r2) * (n - 1) / (n - 2))
    mse    <- mean((obs - pred)^2)
  }
  cat(sprintf("%-28s RMSE: %.4f | MAE: %.4f | R²: %.4f | R² Adj: %.4f\n",
              modele, rmse, mae, r2, r2_adj))
  invisible(c(RMSE = rmse, MAE = mae, MSE = mse, R2 = r2, R2_Adj = r2_adj))
}
